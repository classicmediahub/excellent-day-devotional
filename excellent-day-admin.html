✦ Prayer Wall Table
-- Prayer Wall table CREATE TABLE IF NOT EXISTS prayer_requests ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), user_id TEXT NOT NULL, text TEXT NOT NULL, category TEXT DEFAULT 'General', anonymous BOOLEAN DEFAULT false, pray_count INTEGER DEFAULT 0, answered BOOLEAN DEFAULT false, created_at TIMESTAMPTZ DEFAULT NOW() ); -- Allow anyone to read all prayer requests CREATE POLICY "Public read prayer_requests" ON prayer_requests FOR SELECT USING (true); -- Allow authenticated users to insert their own CREATE POLICY "Insert own prayer_requests" ON prayer_requests FOR INSERT WITH CHECK (true); -- Allow users to update their own (for answered + pray_count) CREATE POLICY "Update prayer_requests" ON prayer_requests FOR UPDATE USING (true); -- Allow users to delete their own CREATE POLICY "Delete own prayer_requests" ON prayer_requests FOR DELETE USING (auth.uid()::text = user_id OR user_id = 'anonymous'); -- Enable RLS ALTER TABLE prayer_requests ENABLE ROW LEVEL SECURITY;

Run this SQL in your Supabase project → SQL Editor → New Query to create the devotionals table and set up the admin write policy.

-- 1. Create the devotionals table CREATE TABLE IF NOT EXISTS devotionals ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), date DATE UNIQUE NOT NULL, title TEXT NOT NULL, verse TEXT NOT NULL, verse_ref TEXT NOT NULL, reflection_1 TEXT, reflection_2 TEXT, reflection_3 TEXT, prayer TEXT, tags TEXT[], created_at TIMESTAMPTZ DEFAULT NOW(), updated_at TIMESTAMPTZ DEFAULT NOW() ); -- 2. Enable Row Level Security ALTER TABLE devotionals ENABLE ROW LEVEL SECURITY; -- 3. Anyone can READ devotionals (public app) CREATE POLICY "Public read" ON devotionals FOR SELECT USING (true); -- 4. Only authenticated admins can WRITE CREATE POLICY "Admin write" ON devotionals FOR ALL USING (auth.role() = 'authenticated') WITH CHECK (auth.role() = 'authenticated'); -- 5. Auto-update updated_at on edits CREATE OR REPLACE FUNCTION update_updated_at() RETURNS TRIGGER AS $$ BEGIN NEW.updated_at = NOW(); RETURN NEW; END; $$ LANGUAGE plpgsql; CREATE TRIGGER devotionals_updated_at BEFORE UPDATE ON devotionals FOR EACH ROW EXECUTE FUNCTION update_updated_at();
